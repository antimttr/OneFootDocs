import React, { Component } from 'react';
import { GithubPicker } from 'react-color';

/**
 * This component renders a text depending on dynamic input taken
 * from the user. The user can select how many times a text
 * renders and its color. Imported a 3rd party color picker library.
 */
export default class PeterComponent extends Component {

  /**
   * Constructs the components. State will keep track
   * of how many times the text will render and the color
   * @param {*} props Just props
   */
  constructor(props) {
    super(props);
    this.state = {numOfRender: 0, componentColor: '#FFF000'};

    //NOTE: in the future for all components, please do the binding
    //inside the constructor and not anywhere else
    this.handleNumChange = this.handleNumChange.bind(this);
    this.handleSumbit = this.handleSumbit.bind(this);
  }

  /**
   * If the input number changes, propagates the info to the state
   * @param {*} event Triggering event
   * @returns {void} none
   */
  handleNumChange(event) {
    this.setState({numOfRender: event.target.value, 
      componentColor: this.state.componentColor});
  }

  /**
   * If the input color changes, propagates the info to the state
   * @returns {void}
   * @param {*} color the color to change the text into
   */
  handleColorChange = (color) => {
    this.setState({numOfRender: this.state.numOfRender,
      componentColor: color.hex});
  }

  /**
   * @returns {void} nothing
   * @param {*} event Triggering event
   */
  handleSumbit(event) {
    event.preventDefault();
  }

  /**
   * Renders the component.
   * @returns {*} the JSX of the html that should be rendered
   */
  render() {
    var peterCompArr = [];
    for(let i = 0; i < this.state.numOfRender; i++) {
      peterCompArr.push(
        <p style={{color: this.state.componentColor}}>WE DEVOTE THIS TEXT TO PETER</p>
      );
    }
    return(
      <div>
        <p>Pick a color:</p>
        <GithubPicker color={this.state.componentColor}
          onChangeComplete={this.handleColorChange} />
        <form>
          <label>
            Enter the number of times you want text rendered:  
            <input type="text" 
              value={this.state.numOfRender} 
              onChange={this.handleNumChange}/>
          </label>
        </form>
        <p>{this.state.numOfRender}</p>
        <div>
          {peterCompArr}
        </div>
      </div>
    );
  }
}
