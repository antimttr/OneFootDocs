import React, { Component } from 'react';
import './App.css';
import fire from './fire';
import './HelloWorld';

/**
 * Main class for Hello world Component
 */
class App extends Component {
  /**
   * Default Constructor
   * @param {object} props input
   */
  constructor(props) {
    super(props);
    this.handleSubmit = this.addMessage.bind(this);
    this.state = { messages: [] }; // <- set up react state
  }

  /**
   * Create reference to messages in Firebase Database
   * @returns {void} none
   */
  componentWillMount() {
    const messagesRef = fire.database().ref('messages').orderByKey().limitToLast(100);
    messagesRef.on('child_added', (snapshot) => {
      /* Update React state when message is added at Firebase Database */
      const message = { text: snapshot.val(), id: snapshot.key };
      this.setState({ messages: [message].concat(this.state.messages) });
    });
  }

  /**
   * Adds a message to the database
   * @param {string} e string of message
   * @returns {void} none
   */
  addMessage(e) {
    e.preventDefault(); // <- prevent form submit from reloading the page
    /* Send the message to Firebase */
    fire.database().ref('messages').push(this.inputEl.value);
    this.inputEl.value = ''; // <- clear the input
  }

  /**
   * Renderer
   * @returns {*} JSX
   */
  render() {
    return (
      <div>
        <form onSubmit={this.handleSubmit}>
          <input type="text" ref={(el) => { this.inputEl = el; }} />
          <input type="submit" />
          <ul>
            { /* Render the list of messages */
              this.state.messages.map(message => <li key={message.id}>{message.text}</li>)
            }
          </ul>
        </form>
      </div>
    );
  }
}


export default App;
